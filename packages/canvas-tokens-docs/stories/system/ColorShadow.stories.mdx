## Usage

```ts
// styles.ts
import {system} from '@workday/canvas-tokens-web';

const styles = {
  boxShadow: `0 0 0 var(${system.color.shadow['1']})`,
};
```

```css
/* styles.css */
.card {
  box-shadow: 0 0 0 var(--cnvs-sys-color-shadow-1);
}
```
