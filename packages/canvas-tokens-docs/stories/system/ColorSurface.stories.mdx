## Usage

```ts
// styles.ts
import {system} from '@workday/canvas-tokens-web';

const styles = {
  backgroundColor: `var(${system.color.surface.default})`,
};
```

```css
/* styles.css */
.card {
  background-color: var(--cnvs-sys-color-surface-default);
}
```
